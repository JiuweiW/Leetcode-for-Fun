# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def insertIntoMaxTree(self, root: TreeNode, val: int) -> TreeNode:
        newNode = TreeNode(val)
        node = root
        if node is None:
            return newNode
        if newNode.val > node.val:
            newNode.left = root
            return newNode
        flag = False
        while node.right is not None:
            if node.right.val > newNode.val:
                node = node.right
            else:
                flag = True
                temp = node.right
                node.right = newNode
                newNode.left = temp
                break
        if not flag:
            node.right = newNode
        return root

class Solution(object):
    def insertIntoMaxTree(self, root, val):
        """
        :type root: TreeNode
        :type val: int
        :rtype: TreeNode
        """
        if not root or val > root.val:
            r = TreeNode(val)
            r.left = root
            return r
        p, c = root, root.right
        while c and c.val > val: p, c = c, c.right
        p.right = TreeNode(val)
        p.right.left = c
        return root

[1023](https://leetcode.com/problems/binary-string-with-substrings-representing-1-to-n/) Binary String With Substrings Representing 1 To N

```python
class Solution:
    def queryString(self, S: str, N: int) -> bool:
        mem = set([])
        for n in range(N, 0, -1):
            s = bin(n)[2:]
            num1 = int(s+"0", 2)
            if num1 in mem:
                continue
            num2 = int(s+"1", 2)
            if num2 in mem:
                continue
            if not s in S:
                return False
            # print(s)
            mem.add(n)
        return True
```

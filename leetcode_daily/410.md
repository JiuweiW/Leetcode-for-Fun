class Solution:
    def splitArray(self, nums: 'List[int]', m: 'int') -> 'int':
        n = len(nums)
        l = 0
        r = 0
        for i in range(n):
            if l < nums[i]:
                l = nums[i]
            r += nums[i]
        
        # binary search the results in [l, r]
        ans = r
        while l <= r:
            mid = l + (r - l)//2
            s = 0
            cnt = 1
            for i in range(n):
                if s + nums[i] > mid:
                    cnt += 1
                    s = nums[i]
                else:
                    s += nums[i]
            if cnt <= m:
                ans = min(ans, mid)
                r = mid - 1
            else:
                l = mid + 1
        return ans

class Solution {
    public int splitArray(int[] nums, int m) {
        int n = nums.length;
        int[][] f = new int[n + 1][m + 1];
        int[] sub = new int[n + 1];
        for (int i = 0; i <= n; i++) {
            for (int j = 0; j <= m; j++) {
                f[i][j] = Integer.MAX_VALUE;
            }
        }
        for (int i = 0; i < n; i++) {
            sub[i + 1] = sub[i] + nums[i];
        }
        f[0][0] = 0;
        for (int i = 1; i <= n; i++) {
            for (int j = 1; j <= m; j++) {
                for (int k = 0; k < i; k++) {
                    f[i][j] = Math.min(f[i][j], Math.max(f[k][j - 1], sub[i] - sub[k]));
                }
            }
        }
        return f[n][m];        
    }
}
